    # 1.Vue3如何全局配置axios

    在main.js入口文件中，通过app.config.globalProperties全局挂载axios

    import axios from 'axios'

const app = createApp(App)

axios.defaults.baseURL = 'http://www.111.cn'
app.config.globalProperties.$http = axios

    # 2.路由概念

    前端路由指Hash地址与组件之间的对应关系 再SPA项目中，不同功能之间的切换，要依赖于前端路由完成

    # vue-router版本

    vue-router目前有3.x的版本和4.x的版本，其中：

    • vue-router 3.x只能结合vue2进行使用
    • vue-router 4.x只能结合vue3进行使用

    # 3.vue-router基本用法

    导入组件后使用router-link标签来声明路由链接， 并使用router-view标签来声明路由占位符

    <router-link to="/home">首页</router-link>

//声明占位符
<router-view></router-view>

    # 创建路由模块

    在项目中创建router.js路由模块

    // createRouter方法用于创建路由的实例对象
// createWebHashHistory用于指定路由的工作模式（hash模式）
import { createRouter, createWebHashHistory } from 'vue-router'

import Home from './Home.vue'

const router = createRouter([
  //通过history属性指定路由的工作模式
  history: createWebHashHistory(),
  //通过routes数组，指定路由规则
  routes: [
    {path: '/home',component: Home}
  ]
])
export default router

    # 路由重定向

    路由重定向：用户在访问地址A的时候，强制用户跳转到地址C 通过路由规则的redirect属性，指定一个新的路由地址

    const router = createRouter([
  history: createWebHashHistory(),
  routes: [
    {path: '/',redirect: '/home'},
    {path: '/home',component: Home}
  ]
])

    # 动态路由匹配

    动态路由：把Hash地址中可变的部分定义为参数项

    {path: '/movie/:id',component: Movie}
    1

    通过动态路由匹配渲染的组件， 可以使用 $route.params 对象访问动态匹配的参数值。

    为了简化路由参数的获取形式，可以在路由规则中开启props传参

    {path: '/movie/:id',component: Movie, props:true}
<template>
<h1>{{id}}</h1>
</template>

export default{
  props:['id']

    # 编程式导航

    通过调用API实现导航的方式，叫做编程式导航 与之对应的，通过点击链接实现导航的方式，叫做声明式导航

    最常用的两个API：

    • this.$router.push('hash地址') 跳转到指定的Hash地址
    • this.$router.go(数值n) 实现导航历史的前进(1)、后退(-1)

    # 5.导航守卫

    全局导航守卫会拦截每个路由规则，控制路由的访问权限

    const router = createRouter({...})
router.beforeEach(fn)
    1
    2

    全局导航守卫的守卫方法中接收3个形参

    router.beforeEach((to,from,next)=>{
  //to 目标路由对象
  //from 当前导航正要离开的路由对象
  //next 放行函数
})
    1
    2
    3
    4
    5

    在守卫方法中如果不声明next形参，则默认允许用户访问每一个路由 在守卫方法中如果声明next形参,则必须调用next()函数，否则不允许用户访问任一个路由

    next函数拥有三种调用方式：

    1. next() 直接放行
    2. next(false) 强制其停留在当前页面
    3. next('/login') 强制其跳转到登陆页面

    结合token控制后台主页访问权限

    router.beforeEach((to,from,next) => {
  const token = localStorage.getItem('token')
  if(to.path === '/main' && !token){
    next('/login')
  } else {
    next()
  }
})
